Создание кнопки


#include <windows.h>


#define ID_BUTTON 3000  //константа идентификатор кнопки


//Объявление функций

BOOL RegClass(WNDPROC, LPCTSTR, UINT);

LRESULT CALLBACK WndProc(HWND, UINT, WPARAM, LPARAM);


//Объявление глобальных переменных

HINSTANCE hInstance;

char szClassName[ ] = "ClassName";


//Главная функция приложения ---------------------------------

int WINAPI WinMain(HINSTANCE hInst, HINSTANCE hPrevInstance, LPSTR lpszCmdLine, int nCmdShow)

{

       //объявление переменных

           MSG msg;

           HWND hwnd;

           hInstance = hInst;

   

       //функция регистрации окна

           if(!RegClass(WndProc, szClassName, COLOR_WINDOW))

               return FALSE;


       //непосредственное создание окна

           hwnd = CreateWindow(

                                  szClassName,

                                  "Демонстрация создания кнопки.",

                                  WS_OVERLAPPEDWINDOW,

                                  CW_USEDEFAULT,

                                 CW_USEDEFAULT,

                                  CW_USEDEFAULT,

                                  CW_USEDEFAULT,

                                  0,0,hInstance, NULL);

   

       //проверка успешности создания окна

           if(!hwnd) return FALSE;

       

       //показать и перерисовать окно

       ShowWindow(hwnd, SW_SHOWMAXIMIZED);

       UpdateWindow(hwnd);

       

       while(GetMessage(&msg, 0, 0, 0)) DispatchMessage(&msg);

       return msg.wParam;

}


//Функция регистрации класса главного окна приложения ----------

BOOL RegClass(WNDPROC Proc, LPCTSTR szName, UINT brBackground)

{

       WNDCLASS wc;


       wc.style = 0;

           wc.cbClsExtra = 0;

           wc.cbWndExtra = 0;

       wc.lpfnWndProc = Proc;

       wc.hInstance = hInstance;

       wc.hIcon = LoadIcon(NULL, IDI_APPLICATION);

       wc.hCursor = LoadCursor(NULL, IDC_ARROW);

       wc.hbrBackground = (HBRUSH)(brBackground + 1);

       wc.lpszMenuName = (LPCTSTR) NULL;

       wc.lpszClassName = szName;

       return (RegisterClass(&wc)!= 0);

}


//Обработка сообщения главного окна. -----------------------------

LRESULT CALLBACK WndProc(HWND hwnd, UINT msg, WPARAM wParam, LPARAM lParam)

{

           static HWND nButton; //описывает дескриптор дочернего окна для кнопки


       switch(msg)

       {

                      case WM_CREATE:

                       {

                       //создание кнопки в главном окне прилодения

                            nButton = CreateWindow(

                                       "button", //имя класса кнопки

                                                      "Пример кнопки",

                                                      WS_CHILD | WS_VISIBLE | BS_DEFPUSHBUTTON,

                                                      10,10,200,25,

                                                     hwnd,

                                                      (HMENU)ID_BUTTON,

                                                      hInstance,

                                                      NULL);

                            return 0;

                       }

               case WM_DESTROY:

               {

                       //закрытие главного окна

                       PostQuitMessage(0);

                       return 0;

               }

       }

       return DefWindowProc(hwnd, msg, wParam, lParam);

}
