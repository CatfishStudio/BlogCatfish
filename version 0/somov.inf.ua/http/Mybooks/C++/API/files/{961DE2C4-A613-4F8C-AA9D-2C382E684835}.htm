Создание главного окна

#include <windows.h>


//Объявление функций

BOOL RegClass(WNDPROC, LPCTSTR, UINT);

LRESULT CALLBACK WndProc(HWND, UINT, WPARAM, LPARAM);


//Объявление глобальных переменных

HINSTANCE hInstance;

char szMainClass[ ] = "MainClass";

char szTitle[ ] = "Демонстрация создания окна приложения.";


//Главная функция приложения ---------------------------------

int WINAPI WinMain(HINSTANCE hInst, HINSTANCE hPrevInst, LPSTR lpszCmdLine, int nCmdShow)

{

       //объявление переменных

       MSG msg;

       HWND hwnd;

       hInstance = hInst;

       

       //функция регистрации окна

       if(!RegClass(WndProc, szMainClass, COLOR_DESKTOP))

               return FALSE;

       

       //Размеры окна

       int w = 800;

       int h = 600;

       

       //непосредственное создание окна

       hwnd = CreateWindow(

                       szMainClass,

                       szTitle,

                       WS_POPUPWINDOW | WS_CAPTION | WS_MINIMIZEBOX | WS_VISIBLE,

                       0,

                       0,

                       w,

                       h,

                       0,

                       0,

                       hInstance,

                       NULL);

       

       //проверка успешности создания окна

       if(!hwnd) return FALSE;

       

       while(GetMessage(&msg, 0, 0, 0)) DispatchMessage(&msg);

       return msg.wParam;

}


//Функция регистрации класса главного окна приложения ----------

BOOL RegClass(WNDPROC Proc, LPCTSTR szName, UINT brBackground)

{

       WNDCLASS wc;


       wc.style = CS_DBLCLKS;

          wc.cbClsExtra = 0;

           wc.cbWndExtra = 0;

       wc.lpfnWndProc = Proc;

       wc.hInstance = hInstance;

       wc.hIcon = LoadIcon(NULL, IDI_APPLICATION);

       wc.hCursor = LoadCursor(NULL, IDC_ARROW);

       wc.hbrBackground = (HBRUSH)(brBackground + 1);

       wc.lpszMenuName = (LPCTSTR) NULL;

       wc.lpszClassName = szName;

       return (RegisterClass(&wc)!= 0);

}


//Обработка сообщения главного окна. -----------------------------

LRESULT CALLBACK WndProc(HWND hwnd, UINT msg, WPARAM wParam, LPARAM lParam)

{

       switch(msg)

       {

               case WM_DESTROY:

               {

                       //закрытие главного окна

                       PostQuitMessage(0);

                       return 0;

               }

       }

       return DefWindowProc(hwnd, msg, wParam, lParam);

}
