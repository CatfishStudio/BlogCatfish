Классы.

Классы это расширенные структуры объектно-ориентированного программирования.

Классы могут содержать переменные, процедуры и функции доступные во всей программе, для более удобной работы.

Классы могут храниться во внешних файлах с расширением  *.h  и быть подключенными благодаря директиве препроцессора #include < *.h >

Синтаксис объявления класса следующий:

Class <имя_класса> : <список классов - родителей>

{

               Public:                // доступно всем

                       < данные, методы, свойства, события >

               __published                // видны в Инспекторе объекта и изменяемы

                       < данные, свойства >

               Protected:                // доступно только потомкам

                       < данные, методы, свойства, события >

               Private :                // доступно только в классе

                       < данные, методы, свойства, события >

} < список переменных > ;


Для того чтобы хранить классы во внешнем файле понадобится создать файл с расширением *.h в котором будет озаглавлен класс и файл с расширением *.cpp в котором он будет описан.

Пример файл MyClass.h:

       #ifndef MYCLASS_H

       #define MYCLASS_H

       Class MyClass

       {

               public :

                       // создание класса

                       Message( );

                       // разрушение класса

                       ~Message( );

       };

       #endif  //MYCLASS_H

       ------------------------------------------------------------------------------------------------------------

Пример файла MyClass.cpp

       #include "stdio.h"

       // создание класса

       MyClass::Message( )

       {

               // оператор

       }


       // разрушение класса

       MyClass::~Message( )

       {

               // оператор

             }
