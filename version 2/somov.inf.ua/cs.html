Синтаксис языка C#: Структура
- Подобие класса но более облегченный вариант.
public struct Rectangle
{
   public int Width;
   public int Height;
   // конструктор
   public Rectangle(int _w, int _h)
   {
      this.Width = _w;
      this.Height = _h;
   }
   // метод
   public int Area()
   {
      return Width * Height;
   }
}

- Работа со структурой
Rectangle rect = new Rectangle(2, 3);
rect.Width = 10;
rect.Area();
