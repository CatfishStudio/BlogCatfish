Unity3D: Перемещение объекта к назначенной целе.
transform. Перемещение объекта к назначенной целе.
public Transform target;
public int moveSpeed;
public int rotationSpeed;
private Transform _myTransform;

void Awake(){
   _myTransform = transform;
}

void Update(){
   //рисуем дебажную линию к цели
   Debug.DrawLine(target.position, _myTransform.position, Color.yellow);
   //поворот в сторону цели
   _myTransform.rotation = Quaternion.Slerp(_myTransform.rotation, Quaternion.lookRotation(target.position - _myTransform.position), rotationSpeed * Time.deltaTime);
   //движение к цели
   _myTransform.position += _myTransform.forward * moveSpeed * Time.deltaTime;
   //Определение дистанции и локальная замена цели
   if(Vector3.Distance(target.position, _myTransform.position) < 2.0f){
      GameObject go = GameObject.FindGameObjectWithTag("Player1");
      target = go.transform;
   }
}

/* Глобальная замена цели */
void OnMouseDown(){
   GameObject go = GameObject.FindGameObjectWithTag("Player2");
   TargetScript TS = GameObject.FindObjectOfType(typeof(TargetScript)) as TargetScript;
   TS.target = go.transform;
}
